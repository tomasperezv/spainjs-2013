Display a Google Maps object

Make sure you add the Javascript right before the ending </body> element.

The Manifest

"RequiredPermissions": [
    "https://*.googleapis.com",
    "https://*.gstatic.com",
    "https://*.gmaptiles.co.kr",
    "https://earthbuilder.google.com",
    "https://earthbuilder.googleapis.com"
]

The Javascript

<script src="https://maps.googleapis.com/maps/api/js?sensor=false"></script>
<script>
    var map_object;
    function initialize() {
        var myOptions = {
            zoom: 10,
            center: new google.maps.LatLng(40.7415,-74.0045),
            mapTypeId: google.maps.MapTypeId.ROADMAP
        };
        map_object = new google.maps.Map(document.getElementById('map_object'), myOptions);
    }
    google.maps.event.addDomListener(window, 'load', initialize);
</script>

The HTML

<div id="map_object" style="width:800px;height:400px;"></div>
